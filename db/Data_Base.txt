CREATE database scala;
USE scala;

CREATE TABLE usuario (
    id_usuario Int auto_increment PRIMARY KEY,
    nome Varchar(30),
    email Varchar(40),
    senha Varchar(50),
    tema Varchar(30) DEFAULT 'css/light.css',
    dark Varchar(30) DEFAULT 'css/dark.css',
    imagem Varchar (100),
    horario DATETIME
);
